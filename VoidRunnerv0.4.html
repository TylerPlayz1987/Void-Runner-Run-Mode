<script>

window.onload = function() {

    let audioCtx, masterVol = 0.5, frameCount = 0, running = false, isPaused = false, currentLevel = 1, lastCheckpoint = 1, deathCount = 0;

    const canvas = document.getElementById('canvas'), ctx = canvas.getContext('2d'), container = document.getElementById('container'), cpNotif = document.getElementById('cp-notif');

    canvas.width = 800; canvas.height = 400;

    let bestLevel = localStorage.getItem('core_best_v20') || 1;

    document.getElementById('best').textContent = bestLevel;

    let shieldItem = null, wells = [];

    let currentTheme = 'classic';

    const themes = {

        classic: { bg: '#050505', plat: '#333', player: '#fff', hazards: '#f33' },

        sunny: { bg: '#87CEEB', plat: '#fff', player: '#FFFF00', hazards: '#f33' },

        moony: { bg: '#000000', plat: '#D8BFD8', player: '#FFF', hazards: '#FFFF33', stars: true },

        toybox: { bg: 'linear-gradient(#ffeb3b, #fbc02d)', plat: '#f44336', player: '#2196f3', hazards: '#4caf50', stars: false },

        deepsea: { bg: 'linear-gradient(#0077be, #001a33)', plat: '#2c3e50', player: '#f1c40f', hazards: '#e74c3c', stars: false }

    };

    const stars = Array.from({length: 50}, () => ({x: Math.random() * 800, y: Math.random() * 400, s: Math.random() * 2}));

    let platforms = [], hazards = [], keys = {};

    const player = {

        x: 50, y: 200, w: 20, h: 20, vx: 0, vy: 0, speed: 5, runSpeed: 8, jump: -12,

        grounded: false, trail: [], hasShield: false, invincibleTimer: 0,

        spaghettified: false, deathScale: 1, deathRotate: 0,

        isBeingRescued: false, rescueTarget: null

    };

    let goal = { x: 0, y: 0, w: 40, h: 40 };

    function sfx(f, t, d, v, slide = 0) {

        try {

            if (!audioCtx) return;

            if (audioCtx.state === 'suspended') audioCtx.resume();

            const o = audioCtx.createOscillator(), g = audioCtx.createGain();

            o.type = t; o.frequency.setValueAtTime(f, audioCtx.currentTime);

            if (slide) o.frequency.exponentialRampToValueAtTime(slide, audioCtx.currentTime + d);

            g.gain.setValueAtTime(v * masterVol, audioCtx.currentTime);

            g.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + d);

            o.connect(g); g.connect(audioCtx.destination); o.start(); o.stop(audioCtx.currentTime + d);

        } catch(e) {}

    }

    const play = {

        jp: () => sfx(180, 'square', 0.1, 0.08, 450),

        wn: () => { sfx(523, 'triangle', 0.1, 0.1); setTimeout(() => sfx(659, 'triangle', 0.1, 0.1), 80); },

        di: () => sfx(150, 'sawtooth', 0.4, 0.1, 40),

        cp: () => { sfx(300, 'sine', 1.2, 0.2, 1400); sfx(450, 'sine', 1.2, 0.1, 1600); },

        well: () => sfx(60, 'sine', 0.8, 0.15, 10),

        grab: () => sfx(400, 'sawtooth', 0.2, 0.1, 100)

    };

    function generateLevel() {

        shieldItem = null; wells = [];

        platforms = [{ x: 0, y: 350, w: 300, h: 50, isPhase: false }];

        hazards = [];

        let lx = 300, ly = 350;

        for (let i = 0; i < 4 + currentLevel; i++) {

            let gp = 110 + Math.random() * 80, vd = (Math.random() - 0.5) * 140;

            if (currentLevel >= 15 && Math.random() < 0.3) hazards.push({ x: lx + gp/2 - 10, y: 100, w: 15, h: 80, s: 2, dir: 1, r: 200, sy: 50 });

            lx += gp; ly = Math.max(120, Math.min(340, ly + vd));

            let isG = (i === 3 + currentLevel);

            let ph = (currentLevel >= 30 && !isG && !platforms[platforms.length-1].isPhase && Math.random() < 0.35);

            let isSink = (currentLevel >= 50 && !isG && !ph && Math.random() < 0.3);

            platforms.push({ x: lx, y: ly, w: 120, h: 15, isPhase: ph, isSinking: isSink, isTouched: false,

                hasSpike: (currentLevel >= 10 && !isG && !ph && !isSink && Math.random() < 0.3),

                moveRange: (currentLevel >= 20 && !isG && Math.random() < 0.4 ? 50 + Math.random() * 50 : 0), startX: lx, moveDir: 1, moveSpeed: 1.5, hasSeeker: (currentLevel >= 40 && !isG && !ph && !isSink && Math.random() < 0.3), seekerX: 0, seekerDir: 1 });

            let lastP = platforms[platforms.length - 1];

            if (currentLevel >= 10 && !isG && Math.random() < 0.25) wells.push({ x: lx + 60, y: ly - 100, r: 110, core: (currentLevel >= 55 ? 10 : 0) });

            if (currentLevel >= 35 && !shieldItem && !isG && !lastP.isPhase && Math.random() < 0.20) shieldItem = { x: lastP.x + lastP.w / 2 - 10, y: lastP.y - 30, w: 20, h: 20, collected: false };

            if (isG) { goal.x = lx + 40; goal.y = ly - 45; }

        }

    }

    function die() {

        if (player.spaghettified) return;

        deathCount++; document.getElementById('deaths').textContent = deathCount;

        currentLevel = lastCheckpoint; document.getElementById('lvl').textContent = currentLevel;

        play.di(); container.style.borderColor = '#f44';

        setTimeout(() => container.style.borderColor = '#333', 200);

        generateLevel(); player.x = 50; player.y = 330; player.vx = 0; player.vy = 0; player.trail = [];

        player.hasShield = false; player.invincibleTimer = 0;

        player.spaghettified = false; player.deathScale = 1; player.deathRotate = 0;

        player.isBeingRescued = false;

    }

    function findSafePlatform() {

        let safest = platforms[0];

        let minDist = Infinity;

        platforms.forEach(p => {

            if (!p.isPhase && !p.hasSpike && !p.hasSeeker) {

                let d = Math.abs(p.x - player.x);

                if (d < minDist) { minDist = d; safest = p; }

            }

        });

        return safest;

    }

    function win() {

        currentLevel++; document.getElementById('lvl').textContent = currentLevel;

        if (currentLevel % 5 === 0) {

            lastCheckpoint = currentLevel; play.cp(); cpNotif.style.opacity = "1";

            container.style.borderColor = '#0f0'; container.style.boxShadow = '0 0 30px #0f0';

            setTimeout(() => { container.style.borderColor = '#333'; container.style.boxShadow = 'none'; cpNotif.style.opacity = "0"; }, 1500);

        } else play.wn();

        if (currentLevel > bestLevel) { bestLevel = currentLevel; document.getElementById('best').textContent = bestLevel; localStorage.setItem('core_best_v20', bestLevel); }

        player.x = 50; player.y = 200; player.vx = 0; player.vy = 0; player.invincibleTimer = 0;

        player.isBeingRescued = false;

        generateLevel();

    }

    function update() {

        if (!running || isPaused) return;

        frameCount++;

        if (player.isBeingRescued) {

            let targetX = player.rescueTarget.x + player.rescueTarget.w / 2 - player.w / 2;

            let targetY = player.rescueTarget.y - player.h - 5;

            player.x += (targetX - player.x) * 0.08;

            player.y += (targetY - player.y) * 0.08;

            if (Math.abs(player.x - targetX) < 5 && Math.abs(player.y - targetY) < 5) {

                player.isBeingRescued = false;

                player.vx = 0; player.vy = 0;

                player.invincibleTimer = 60;

            }

            return;

        }

        if (player.invincibleTimer > 0) player.invincibleTimer--;

        if (player.spaghettified) { player.deathScale *= 0.96; player.deathRotate += 0.15; if (player.deathScale < 0.05) { player.spaghettified = false; die(); } return; }

        if (keys['ShiftLeft'] || keys['ShiftRight']) { player.trail.push({x: player.x, y: player.y}); if (player.trail.length > 8) player.trail.shift(); } else { player.trail = []; }

        let s = (keys['ShiftLeft'] || keys['ShiftRight']) ? player.runSpeed : player.speed;

        player.vx = (keys['ArrowLeft'] || keys['KeyA']) ? -s : (keys['ArrowRight'] || keys['KeyD']) ? s : player.vx * 0.85;

        player.vy += 0.6;

        let dead = false, suckedIn = false;

        wells.forEach(w => {

            let dx = w.x - (player.x + 10), dy = w.y - (player.y + 10), dist = Math.sqrt(dx*dx + dy*dy);

            if (dist < w.r) {

                let force = (1 - dist/w.r) * 1.1; player.vx += (dx/dist) * force; player.vy += (dy/dist) * force;

                if (w.core > 0 && dist < w.core) {

                    if (player.hasShield) { player.hasShield = false; player.invincibleTimer = 120; player.vy = -16; play.di(); }

                    else if (player.invincibleTimer <= 0) { suckedIn = true; dead = true; }

                }

            }

        });

        if ((keys['ArrowUp'] || keys['KeyW'] || keys['Space']) && player.grounded) { player.vy = player.jump; player.grounded = false; play.jp(); }

        let nX = player.x + player.vx, nY = player.y + player.vy;

        player.grounded = false;

        if (shieldItem && !shieldItem.collected) { if (nX < shieldItem.x + shieldItem.w && nX + player.w > shieldItem.x && nY < shieldItem.y + shieldItem.h && nY + player.h > shieldItem.y) { shieldItem.collected = true; player.hasShield = true; play.wn(); } }

        if (player.invincibleTimer <= 0 && !dead) {

            for (let h of hazards) { if (nX < h.x + h.w && nX + player.w > h.x && nY < h.y + h.h && nY + player.h > h.y) { dead = true; break; } }

            for (let p of platforms) { if (p.hasSpike || p.hasSeeker) { let sx = p.hasSpike ? p.x + 52.5 : p.x + p.seekerX; if (nX + player.w > sx && nX < sx + 15 && nY + player.h > p.y - 12 && nY < p.y) { dead = true; break; } } }

        }

       

        if (player.y > 400) {

            if (player.hasShield) {

                player.hasShield = false;

                player.isBeingRescued = true;

                player.rescueTarget = findSafePlatform();

                play.grab();

                return;

            } else {

                die(); return;

            }

        }

        if (dead) { if (player.hasShield && !suckedIn) { player.hasShield = false; player.invincibleTimer = 120; player.vy = -8; play.di(); } else if (suckedIn) { player.spaghettified = true; play.well(); return; } else { die(); return; } }

        hazards.forEach(h => { if (h.r) { h.sy += h.s * h.dir; if (h.sy > h.r || h.sy < 50) h.dir *= -1; h.y = h.sy; } });

        platforms.forEach(p => {

            let cycle = frameCount % 300, solid = !p.isPhase || (cycle < 240);

            let deltaX = 0; if (p.moveRange > 0) { let oldX = p.x; p.x += p.moveSpeed * p.moveDir; if (p.x > p.startX + p.moveRange || p.x < p.startX - p.moveRange) p.moveDir *= -1; deltaX = p.x - oldX; }

            if (p.isSinking && p.isTouched) p.y += 2.5;

            if (p.hasSeeker) { p.seekerX += 2.5 * p.seekerDir; if (p.seekerX > p.w - 15 || p.seekerX < 0) p.seekerDir *= -1; }

            if (solid && nX + player.w > p.x && nX < p.x + p.w && player.y + player.h <= p.y && nY + player.h >= p.y) { player.vy = 0; nY = p.y - player.h; player.grounded = true; if (p.isSinking) p.isTouched = true; if (deltaX !== 0) nX += deltaX; }

        });

        player.x = nX; player.y = nY;

        if (player.x + player.w > goal.x && player.x < goal.x + goal.w && player.y + player.h > goal.y && player.y < goal.y + goal.h) { win(); return; }

    }

    function drawPlayer(xPos, yPos, color) {

        ctx.save();

        if (player.spaghettified) {

            let targetWell = wells.find(w => Math.abs(w.x - (player.x + 10)) < w.r);

            if (targetWell) { let camX = player.x - 150; ctx.translate(targetWell.x - camX, targetWell.y); ctx.rotate(player.deathRotate); ctx.scale(player.deathScale, player.deathScale * 3); xPos = -10; yPos = -10; }

        }

        if (player.invincibleTimer > 0) { ctx.shadowBlur = 15; ctx.shadowColor = "#fff"; ctx.globalAlpha = 0.5 + Math.sin(frameCount/5) * 0.3; }

       

        // --- IMPROVED PIXEL HAND ---

        if (player.isBeingRescued) {

            ctx.fillStyle = "#e0ac69"; // Hand flesh tone

            let hX = xPos - 5;

            let hY = yPos - 5;

           

            // Wrist/Arm

            ctx.fillRect(hX + 10, hY + 25, 10, 20);

            // Palm

            ctx.fillRect(hX + 2, hY + 10, 26, 15);

           

            // Fingers (5 fingers)

            ctx.fillStyle = "#d29958"; // Shade for fingers

            // Thumb

            ctx.fillRect(hX - 2, hY + 15, 6, 6);

            // Index

            ctx.fillRect(hX + 4, hY - 4, 4, 14);

            // Middle

            ctx.fillRect(hX + 10, hY - 8, 4, 18);

            // Ring

            ctx.fillRect(hX + 16, hY - 6, 4, 16);

            // Pinky

            ctx.fillRect(hX + 22, hY, 4, 10);

        }

        if (currentTheme === 'moony') { ctx.fillStyle = color; ctx.beginPath(); ctx.arc(xPos + 10, yPos + 10, 10, 0, Math.PI * 2); ctx.fill(); ctx.globalCompositeOperation = 'destination-out'; ctx.beginPath(); ctx.arc(xPos + 16, yPos + 6, 10, 0, Math.PI * 2); ctx.fill(); ctx.globalCompositeOperation = 'source-over'; }

        else if (currentTheme === 'sunny') { ctx.fillStyle = color; const centerX = xPos + 10, centerY = yPos + 10; ctx.beginPath(); ctx.arc(centerX, centerY, 7, 0, Math.PI * 2); ctx.fill(); for (let i = 0; i < 8; i++) { ctx.beginPath(); let angle = (i / 8) * (Math.PI * 2); ctx.moveTo(centerX + Math.cos(angle) * 7, centerY + Math.sin(angle) * 7); ctx.lineTo(centerX + Math.cos(angle) * 11, centerY + Math.sin(angle) * 11); ctx.lineWidth = 2; ctx.strokeStyle = color; ctx.stroke(); } }

        else if (currentTheme === 'toybox') { ctx.fillStyle = color; ctx.fillRect(xPos + 2, yPos + 6, 16, 14); ctx.fillRect(xPos + 5, yPos + 1, 10, 5); ctx.fillRect(xPos + 9, yPos, 2, 2); ctx.fillStyle = "#fff"; ctx.fillRect(xPos + 7, yPos + 2, 2, 2); ctx.fillRect(xPos + 11, yPos + 2, 2, 2); }

        else if (currentTheme === 'deepsea') { ctx.fillStyle = color; ctx.fillRect(xPos, yPos + 5, 20, 12); ctx.fillRect(xPos + 12, yPos + 1, 2, 4); ctx.fillRect(xPos + 12, yPos + 1, 4, 2); ctx.fillStyle = "#fff"; ctx.beginPath(); ctx.arc(xPos + 8, yPos + 11, 3, 0, Math.PI * 2); ctx.fill(); }

        else { ctx.fillStyle = color; ctx.fillRect(xPos, yPos, player.w, player.h); }

        ctx.restore();

    }

    function draw() {

        const t = themes[currentTheme]; container.style.background = t.bg;

        ctx.clearRect(0, 0, canvas.width, canvas.height);

        const camX = player.x - 150;

        wells.forEach(w => {

            ctx.beginPath(); ctx.arc(w.x - camX, w.y, w.r, 0, Math.PI*2);

            let grad = ctx.createRadialGradient(w.x-camX, w.y, 0, w.x-camX, w.y, w.r);

            if (w.core > 0) { grad.addColorStop(0, "rgba(128, 0, 255, 0.5)"); grad.addColorStop(0.3, "rgba(75, 0, 130, 0.3)"); } else { grad.addColorStop(0, "rgba(0, 128, 255, 0.5)"); grad.addColorStop(0.3, "rgba(0, 75, 130, 0.3)"); }

            grad.addColorStop(1, "transparent"); ctx.fillStyle = grad; ctx.fill();

            if (w.core > 0) { ctx.beginPath(); ctx.arc(w.x - camX, w.y, w.core + Math.sin(frameCount/5)*2, 0, Math.PI*2); ctx.fillStyle = "#000"; ctx.fill(); ctx.strokeStyle = "#80f"; ctx.lineWidth = 3; ctx.stroke(); }

            for(let i=0; i<3; i++) { ctx.beginPath(); ctx.arc(w.x-camX, w.y, (w.core || 10) + 10 + i*15, frameCount/20 + i, frameCount/20 + i + 1); ctx.strokeStyle = w.core > 0 ? "rgba(128,0,255,0.2)" : "rgba(0,128,255,0.2)"; ctx.stroke(); }

        });

        if (shieldItem && !shieldItem.collected) { ctx.strokeStyle = "#0cf"; ctx.lineWidth = 3; ctx.strokeRect(shieldItem.x - camX, shieldItem.y, 20, 20); ctx.fillStyle = "rgba(0, 200, 255, 0.5)"; ctx.fillRect(shieldItem.x - camX + 5, shieldItem.y + 5, 10, 10); }
